
  model{
    for(i in 1:N){
      for(j in 1:2){
        y[i,j] ~ dbin(pie[i,j],n[i,j])
        logit(pie[i,j]) <- mu[i] + (2*j - 3) * delta[i]/2
        }
      mu[i] ~ dnorm(mu0, tau.mu)
      delta[i] ~ dnorm(delta0, tau.delta)
      }
    mu0 ~ dnorm(0, 1/3.36)
    tau.mu ~ dunif(0, 1/3.36)
    delta0 ~ dnorm(0, 1/2.17)
    tau.delta ~ dunif(0, 1/2.17)
    }
